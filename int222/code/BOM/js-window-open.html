BOM - open and close window Syntax

	  messageWindow=window.open('','window target name','window properties');
	  messageWindow.document.write(content); 
	  messageWindow.document.close(); // closes the output stream to the document.

	  window.open('url','window target name','window properties');
   

	  myWindow = window.open('url','window target name','window properties');
	  myWindow.close()
