Week 2 - Moment.js ▲

Working with Moment.js

Run the following code in Firefox Scratchpad: the Execute menu -> Reload And Run

let hireDate = "2010-11-07T04:00:00.000Z"; // Zsa zsa Mannering's hire date in our "Employees" Collection

let mDate = moment(hireDate); // create a new "moment" object

mDate.utc(); // switch to "UTC" mode

// display the UTC date
let mDate1 = mDate.format('LLLL'); // mDate1:  Sunday, November 7, 2010 4:00 AM
console.log("mDate1: ", mDate1);

mDate.local(); // switch to "local" mode

// display a localized date (now offset to Local Time)
let mDate2 = mDate.format('LLLL'); // mDate2:  Sunday, November 7, 2010 12:00 AM
console.log("mDate2: ", mDate2);
	
moment.locale("fr-CA");  // Try out Quebec French
mDate = moment(hireDate);

mDate.utc();
mDate1 = mDate.format('LLLL');
console.log("mDate1: ", mDate1); // mDate1:  dimanche 7 novembre 2010 04:00

mDate.local();
mDate2 = mDate.format('LLLL');
console.log("mDate2: ", mDate2); // mDate2:  dimanche 7 novembre 2010 00:00

let mDate = moment("2010-11-07T04:00:00.000Z"); // create a new "moment" object

let mDate19 = mDate.fromNow(); // 7 years ago
console.log("mDate19: ", mDate19);

let mDate21 = mDate.diff(moment([2010,0,1]), 'days'); // 310
let mDate22 = mDate.diff(moment([2010,0,1]), 'months'); // 10
let mDate23 = mDate.diff(moment([2010,0,1]), 'years'); // 0
console.log("mDate21: ", mDate21);
console.log("mDate22: ", mDate22);
console.log("mDate23: ", mDate23);

let mDate24 = mDate.toISOString();
let mDate25 = mDate.daysInMonth(); // 30
console.log("mDate24: ", mDate24);
console.log("mDate25: ", mDate25);

Manipulate Options:

Run the following code in Firefox Scratchpad: the Execute menu -> Reload And Run

let mDate = moment("2010-11-07T04:00:00.000Z"); // create a new "moment" object

mDate.add(5, 'days'); // Using: .format("LLLL") => Friday, November 12, 2010 12:00 AM
console.log("mDate.format(\"LLLL\"): ", mDate.format("LLLL"));

mDate.subtract(5, 'days'); // Using: .format("LLLL") => Sunday, November 7, 2010 12:00 AM
console.log("mDate.format(\"LLLL\"): ", mDate.format("LLLL"));

let mDate27 = mDate.isBefore(moment([2009,11,7])); // false
console.log("mDat27e: ", mDate27);

let mDate28 = mDate.isSame(moment([2010,10,8])); // false
let mDate29 = mDate.isSame(moment([2010,10,8]), "month"); // true
console.log("mDate28: ", mDate28);
console.log("mDate29: ", mDate29);

let mDate30 = mDate.isBetween(moment([2010,0,1]), moment()); // true
console.log("mDate30: ", mDate30);

let mDate31 = mDate.isDST(); // true
console.log("mDate31: ", mDate31);

let mDate32 = mDate.isLeapYear(); // false
console.log("mDate32: ", mDate32);
